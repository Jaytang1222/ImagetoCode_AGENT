def agent1_code_generation(input_image_path, output_code_path="demo1.py"):
    """
    输入：目标图表图片路径
    输出：生成绘图代码并保存
    模型：Qwen-VL-Max (支持视觉输入)
    """
    print(f"[Agent 1] 正在分析图片 {input_image_path} 并生成代码...")
    messages = [
        {
            "role": "user",
            "content": [
                {"image": f"file://{os.path.abspath(input_image_path)}"},
                {"text": "请分析这张科学统计图表，编写一段 Python 代码 (使用 matplotlib 或 seaborn) 来完全复现这张图表。包括数据、颜色、标签、标题等。只输出代码，不要 markdown 标记。"}

    ]

        # 清理可能存在的 markdown 标记
        code_content = code_content.replace("", "").replace("", "").strip()
        with open(output_code_path, "w", encoding="utf-8") as f:

        print(f"[Agent 1] 代码已保存至 {output_code_path}")
        return output_code_path
    else:
        print(f"[Agent 1] 调用失败：{response.code}, {message}")
        return None

任务二：多维验证器详细实现步骤

验证器是 Agent 4 的核心组成部分。我们需要从颜色一致性、文本一致性、整体结构相似性三个维度进行量化评分。

实现思路
颜色一致性：使用 OpenCV 计算两张图片的 HSV 直方图相关性。
文本一致性：使用 Qwen-VL 提取两张图中的文字内容，计算文本相似度（Levenshtein 距离）。
整体结构相似性：使用 OpenCV 的 SSIM (Structural Similarity Index) 算法。

验证器代码实现

python
import cv2
import numpy as np
from skimage.metrics import structural_similarity as ssim
import difflib # 需要安装：pip install scikit-image

class MultiDimensionalValidator:
    def init(self, threshold=0.85):
        """
        初始化验证器
        :param threshold: 综合得分达标阈值 (0-1)
        """
        self.threshold = threshold

    def check_color_consistency(self, img1_path, img2_path):
        """
        维度 1: 颜色一致性
        方法：计算 HSV 直方图相关性
        """
        img1 = cv2.imread(img1_path)
        img2 = cv2.imread(img2_path)
        if img1 is None or img2 is None:
            return 0.0
        
        # 转换为 HSV 色彩空间
        hsv1 = cv2.cvtColor(img1, cv2.COLOR_BGR2HSV)
        hsv2 = cv2.cvtColor(img2, cv2.COLOR_BGR2HSV)
        
        # 计算直方图
        hist1 = cv2.calcHist([hsv1], [0, 1], None, [50, 60], [0, 180, 0, 256])
        hist2 = cv2.calcHist([hsv2], [0, 1], None, [50, 60], [0, 180, 0, 256])
        
        # 归一化
        cv2.normalize(hist1, hist1, 0, 1, cv2.NORM_MINMAX)
        cv2.normalize(hist2, hist2, 0, 1, cv2.NORM_MINMAX)
        
        # 计算相关性 (Correlation)
        score = cv2.compareHist(hist1, hist2, cv2.HISTCMP_CORREL)
        print(f"[Validator] 颜色一致性得分：{score:.4f}")
        return score

    def check_text_consistency(self, img1_path, img2_path):
        """
        维度 2: 文本一致性
        方法：调用 Qwen-VL 提取文字，计算文本相似度
        注意：为了节省 API 调用，此处模拟提取，实际需调用 API
        """
        # 实际实现需要调用 Agent 1 类似的 Qwen-VL 接口提取文字
        # 为了演示逻辑，此处假设我们有一个函数 extract_text_via_qwen
        text1 = self._extract_text_via_qwen(img1_path)
        text2 = self._extract_text_via_qwen(img2_path)
        if not text1 or not text2:
            return 0.5 # 无法提取时给中等分
        
        # 计算序列相似度
        ratio = difflib.SequenceMatcher(None, text1, text2).ratio()
        print(f"[Validator] 文本一致性得分：{ratio:.4f}")
        return ratio

    def _extract_text_via_qwen(self, image_path):
        """
        辅助函数：调用 Qwen-VL 提取图中文字
        """
        messages = [
            {
                "role": "user",
                "content": [
                    {"image": f"file://{os.path.abspath(image_path)}"},
                    {"text": "请提取这张图表中所有的文字内容（标题、坐标轴、图例、数值），直接返回文字字符串，用逗号分隔。"}
                ]
            }
        ]
        try:
            response = MultiModalConversation.call(model='qwen-vl-max', messages=messages)
            if response.status_code == 200:
                return response.output.choices[0].message.content
        except:
            pass
        return ""

    def check_structure_similarity(self, img1_path, img2_path):
        """
        维度 3: 整体结构相似性
        方法：SSIM (结构相似性指数)
        """
        img1 = cv2.imread(img1_path, 0) # 灰度图
        img2 = cv2.imread(img2_path, 0)
        if img1 is None or img2 is None:
            return 0.0
        
        # 调整大小以确保一致 (SSIM 要求尺寸相同)
        img2 = cv2.resize(img2, (img1.shape[1], img1.shape[0]))
        
        score, _ = ssim(img1, img2, full=True)
        print(f"[Validator] 结构相似性得分：{score:.4f}")
        return score

    def validate(self, target_image, generated_image):
        """
        综合验证入口
        返回：(是否达标，综合分数，详细报告)
        """
        s_color = self.check_color_consistency(target_image, generated_image)
        s_text = self.check_text_consistency(target_image, generated_image)
        s_struct = self.check_structure_similarity(target_image, generated_image)
        
        # 加权平均 (可根据需求调整权重)
        # 结构最重要，文本次之，颜色再次
        total_score = s_color * 0.2 + s_text * 0.3 + s_struct * 0.5
        is_pass = total_score >= self.threshold
        
        report = {
            "color_score": s_color,
            "text_score": s_text,
            "structure_score": s_struct,
            "total_score": total_score,
            "is_pass": is_pass
        }
        
        # 保存报告
        with open("validation_report.json", "w", encoding="utf-8") as f:
            json.dump(report, f, ensure_ascii=False, indent=4)
            
        return is_pass, total_score, report

使用指南与注意事项

API 成本控制：
    项目大量使用了 qwen-vl-max，这是一个收费较高的模型。建议在调试阶段使用 qwen-vl-plus 或本地部署的 VLM 模型。
    验证器中的文本提取也可以改为本地 OCR（如 PaddleOCR）以降低成本。
代码执行安全：
    exec 或 subprocess 运行 LLM 生成的代码存在安全风险。务必在 Docker 容器或虚拟机中运行此项目，防止生成的代码恶意操作宿主机。
    在 Agent 1 的 Prompt 中，必须明确要求代码将图片保存为指定文件名（如 example.png），否则后续流程无法获取图片。
验证器调优：
    threshold (阈值) 需要根据实际效果调整。如果太严格，会导致无限循环；太宽松则生成质量差。
    颜色一致性对于白色背景的图表可能得分很高，但对于复杂背景可能需要调整 HSV 参数。
异常处理：
    生成的代码可能会报错（语法错误、库缺失）。需要在 run_code_and_save_image 中增加 try-catch 逻辑，如果运行报错，将错误日志传给 Agent 3 进行修复。
